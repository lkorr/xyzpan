Preset XML format (from RESEARCH.md Pattern 5):
APVTS state tag is "XYZPanState" (set in constructor: `apvts(*this, nullptr, "XYZPanState", ...)`).
XML structure:
```xml
<XYZPanState>
  <PARAM id="x" value="0.0"/>
  <PARAM id="y" value="1.0"/>
  ...
</XYZPanState>
```
Generate preset XMLs by: configure params to desired values in Standalone, call `apvts.copyState().createXml()->toString()`.
Since we can't run the app during planning, write the XML manually using known param IDs and defaults.

Preset loading on message thread:
```cpp
void loadPreset(XYZPanProcessor& proc, const char* xmlString) {
    if (auto xml = juce::XmlDocument::parse(juce::String(xmlString)))
        if (xml->hasTagName(proc.apvts.state.getType()))
            proc.apvts.replaceState(juce::ValueTree::fromXml(*xml));
}
```
This is message-thread-safe (apvts.replaceState is called on message thread from onClick lambda).

```cpp
#pragma once
#include <juce_audio_processors/juce_audio_processors.h>
// Forward declare to avoid circular include
class XYZPanProcessor;

namespace XYZPresets {

static const char* kPresetFrontCenter = R"(
<XYZPanState>
  <!-- All params listed explicitly for full state replacement -->
  <PARAM id="x" value="0.0"/>
  <PARAM id="y" value="1.0"/>
  <PARAM id="z" value="0.0"/>
  <PARAM id="r" value="1.0"/>
  ... (all other params at their defaults)
</XYZPanState>
)";

// ... kPresetOverhead, kPresetBehindAndFar, kPresetOrbitingXY, kPresetDeepSpace

struct PresetEntry {
    const char* name;
    const char* xml;
};

static const PresetEntry kFactoryPresets[] = {
    { "Front",    kPresetFrontCenter },
    { "Overhead", kPresetOverhead },
    { "Behind",   kPresetBehindAndFar },
    { "Orbit",    kPresetOrbitingXY },
    { "Space",    kPresetDeepSpace }
};

inline void loadPreset(juce::AudioProcessorValueTreeState& apvts, const char* xmlString) {
    if (auto xml = juce::XmlDocument::parse(juce::String(xmlString)))
        if (xml->hasTagName(apvts.state.getType()))
            apvts.replaceState(juce::ValueTree::fromXml(*xml));
}

} // namespace XYZPresets
```
