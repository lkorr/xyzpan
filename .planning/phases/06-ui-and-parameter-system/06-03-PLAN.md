  ParamID::X          = "x"

  ParamID::VERB_WET     = "verb_wet"
All IDs are lowercase. NEVER pass uppercase string literals ("PAN_X", "VERB_SIZE", etc.)
to SliderAttachment — they will not match any registered parameter and will cause JUCE
asserts and silently unbound knobs. Always use ParamID:: constants.

```cpp
class LFOWaveformButton : public juce::Component {
    int waveform_ = 0; // 0=sine, 1=tri, 2=saw, 3=square
    juce::AudioProcessorValueTreeState::Parameter* waveformParam_ = nullptr;
public:
    void setParam(juce::AudioProcessorValueTreeState& apvts, const juce::String& paramID) {
        waveformParam_ = apvts.getParameter(paramID);
    }
    void paint(juce::Graphics& g) override {
        // Draw waveform using Graphics path (64-point polyline)
        // Colors from AlchemyLookAndFeel: ochre/bronze lines on dark bg
    }
    void mouseUp(const juce::MouseEvent&) override {
        if (waveformParam_ == nullptr) return;
        waveform_ = ((int)waveformParam_->getValue() + 1) % 4;
        waveformParam_->setValueNotifyingHost((float)waveform_);
        repaint();
    }
};
```

Step 2 — `resized()` — layout with kStripH=200:
```cpp
void XYZPanEditor::resized()
{
    auto bounds = getLocalBounds();
    const int stripH = kStripH;

    // GL view occupies everything above the bottom strip
    auto glArea = bounds.removeFromTop(bounds.getHeight() - stripH);
    glView_.setBounds(glArea);

    // Dev panel overlays right 30% of GL view when visible
    const int panelW = static_cast<int>(glArea.getWidth() * 0.30f);
    devPanel_.setBounds(glArea.getRight() - panelW, glArea.getY(), panelW, glArea.getHeight());

    // Dev toggle button: top-left of GL area
    devToggle_.setBounds(glArea.getX() + 4, glArea.getY() + 4, 40, 22);

    // Bottom strip: divide into columns for X/Y/Z/R knobs + Reverb
    // Each position knob column: knob (top 80px) + LFO strip (bottom 120px)
    const int colW     = 100;
    const int knobH    = 80;
    const int lfoH     = stripH - knobH;  // 120px

    auto strip = bounds;  // remaining area after GL view removal
    int x = strip.getX();

    auto placeKnobAndLFO = [&](juce::Slider& knob, juce::Label& label,
                                LFOStrip* lfo) {
        auto col = strip.withLeft(x).withWidth(colW);
        knob.setBounds(col.getX(), col.getY(), colW, knobH - 16);
        label.setBounds(col.getX(), col.getY() + knobH - 16, colW, 16);
        if (lfo != nullptr)
            lfo->setBounds(col.getX(), col.getY() + knobH, colW, lfoH);
        x += colW;
    };

    placeKnobAndLFO(xKnob_, xLabel_, &xLFO_);
    placeKnobAndLFO(yKnob_, yLabel_, &yLFO_);
    placeKnobAndLFO(zKnob_, zLabel_, &zLFO_);
    placeKnobAndLFO(rKnob_, rLabel_, nullptr);  // R knob has no LFO strip

    // Reverb section: 4 knobs after R column
    auto placeVerbKnob = [&](juce::Slider& knob, juce::Label& label) {
        knob.setBounds(x, strip.getY(), colW, knobH - 16);
        label.setBounds(x, strip.getY() + knobH - 16, colW, 16);
        x += colW;
    };
    placeVerbKnob(verbSize_,    verbSizeL_);
    placeVerbKnob(verbDecay_,   verbDecayL_);
    placeVerbKnob(verbDamping_, verbDampingL_);
    placeVerbKnob(verbWet_,     verbWetL_);
}
```
