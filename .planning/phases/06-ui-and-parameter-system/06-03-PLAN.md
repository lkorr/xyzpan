LFOWaveformButton pattern (from RESEARCH.md):
```cpp
class LFOWaveformButton : public juce::Component {
    int waveform_ = 0; // 0=sine, 1=tri, 2=saw, 3=square
    juce::AudioProcessorValueTreeState::Parameter* waveformParam_ = nullptr;
public:
    void setParam(juce::AudioProcessorValueTreeState& apvts, const juce::String& paramID) {
        waveformParam_ = apvts.getParameter(paramID);
    }
    void paint(juce::Graphics& g) override {
        // Draw waveform using Graphics path (64-point polyline)
        // Colors from AlchemyLookAndFeel: ochre/bronze lines on dark bg
    }
    void mouseUp(const juce::MouseEvent&) override {
        if (waveformParam_ == nullptr) return;
        waveform_ = ((int)waveformParam_->getValue() + 1) % 4;
        waveformParam_->setValueNotifyingHost((float)waveform_);
        repaint();
    }
};
```

```cpp
#pragma once
#include <juce_audio_processors/juce_audio_processors.h>

// Clickable mini waveform display. Cycles 0=sine, 1=triangle, 2=saw, 3=square on click.
// Reads/writes LFO_X_WAVEFORM (or Y/Z equivalent) via APVTS parameter.
class LFOWaveformButton : public juce::Component {
public:
    LFOWaveformButton();
    void setParam(juce::AudioProcessorValueTreeState& apvts, const juce::String& waveformParamID);
    void paint(juce::Graphics& g) override;
    void mouseUp(const juce::MouseEvent&) override;
    void parameterChanged();  // call to sync display from APVTS on load

private:
    int waveform_ = 0;
    juce::AudioProcessorParameter* param_ = nullptr;
    static float computeWaveformY(float t, int waveform);  // t in [0,1], returns [-1,1]
    JUCE_DECLARE_NON_COPYABLE_WITH_LEAK_DETECTOR(LFOWaveformButton)
};
```

**ui/LFOWaveformButton.cpp** — Implement:
- `paint()`: draw background (#1A1108), draw 64-point path for the current waveform in bronze (#8B5E2E), slightly brighter when hovered
- `computeWaveformY(t, 0)`: `std::sin(t * 2.0f * juce::MathConstants<float>::pi)`
- `computeWaveformY(t, 1)`: triangle: `1.0f - 4.0f * std::abs(t - 0.5f)` (matches engine LFO formula)
- `computeWaveformY(t, 2)`: sawtooth: `2.0f * t - 1.0f`
- `computeWaveformY(t, 3)`: square: `t < 0.5f ? 1.0f : -1.0f`
- `mouseUp()`: if param_ != nullptr, increment waveform, call `param_->setValueNotifyingHost((float)waveform_)`, repaint
- `setParam()`: store param pointer; read current value to sync waveform_
